616 {
617  int i,j,k;
618  double rij2,rij3,acc;
619  msm4g_zeros(a,n*d);
620  for (i=0;i<n-1;i++)
621  {
622  for (j=i+1;j<n;j++)
623  {
624  rij2=0.0;
625  for (k=0;k<d;k++)
626  rij2 +=(r[d*i+k]-r[d*j+k])*(r[d*i+k]-r[d*j+k]);
627  rij3=sqrt(rij2)*rij2;

633  }
634  }
