575  /* If only the particles are closer than cut-off distance */
576  if (r2 < a2)
577  {

579  smoothedkernelderivative = 1.0/r2 - msm4g_smoothing_gamaprime(r/a,order)/a2;
580  coeff = particleI->m * particleJ->m * smoothedkernelderivative/r;
581  /* f_j = f_j + coeff*rij */
582  msm4g_d3vector_daxpy(&(particleJ->fshort) , &(particleJ->fshort), coeff, &rij);
583  /* f_i = f_i - coeff*rij */
584  msm4g_d3vector_daxpy(&(particleI->fshort) , &(particleI->fshort), coeff, &rij);
585  potential = particleI->m * particleJ->m * smoothedkernel;
586  particleI->potential_short_real += 0.5 * particleJ->m * smoothedkernel;
587  particleJ->potential_short_real += 0.5 * particleI->m * smoothedkernel;
588  }
589  }
590

592 }

792 void msm4g_box_update(SimulationBox *box,LinkedList *list,double margin,double h,double p)
793 {
794  double min[3];
795  double max[3];
796  double oldwidth,newwidth;
797  int i;
798  LinkedListElement *curr ;
799  Particle *particle;
800  for (i=0;i<3;i++)
801  {
802  min[i] = DBL_MAX;
803  max[i] = DBL_MIN;
804  }
805 
806  curr = list->head;
807  while (curr != NULL)
808  {
809  particle = (Particle *) (curr->data);
810  for (i=0; i<3; i++)
811  {
812  if (particle->r.value[i] >= max[i])
813  max[i] = particle->r.value[i];
814  if (particle->r.value[i] <= min[i])
815  min[i] = particle->r.value[i];
816  }
817  curr = curr->next;
818  }
819 
820  /* enclosing the particles tightly */
821  for (i=0;i<3;i++)
822  {
823  /* If the width in ith dimension is effectively zero,
824  * we artificaially add some width (h) so that
825  * a grid can be created.
826  */
827  if (fabs(max[i]-min[i]) < DBL_EPSILON )
828  {
829  box->location.value[i] = min[i]-h*0.50;
830  box->width.value[i] = h;
831  } else
832  {
833  box->location.value[i] = min[i];
834  box->width.value[i] = max[i]-min[i];
835  }
836  }
837 
838  /* enlarging the box <margin> percent. */
839  if (margin > 0.0)
840  {
841  for (i=0;i<3;i++)
842  {
843  oldwidth = box->width.value[i];
844  newwidth = oldwidth*(1.0+margin);
845  box->location.value[i] -= 0.5*(newwidth-oldwidth);
846  box->width.value[i] = newwidth;
847  }
848  }
849 
850  /* make sure the widths are multiple of h */
851  for (i=0;i<3;i++)
852  {
853  oldwidth = box->width.value[i];
854  newwidth = fabs(floor(oldwidth/h)-oldwidth/h) < 0.0001 ? oldwidth : (floor(oldwidth/h)+1)*h;
855  box->width.value[i] = newwidth;
856  box->location.value[i] -= (newwidth-oldwidth)*0.5;
857  }
858 
859  /* add p/2-1 boxes around the boundary */
860  for (i=0;i<3;i++)
861  {
862  oldwidth = box->width.value[i];
863  newwidth = oldwidth + (p-2)*h ;
864  box->width.value[i] = newwidth;
865  box->location.value[i] -= h*(p/2-1);
866  }
867 }
