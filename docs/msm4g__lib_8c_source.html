163 double msm4g_force_short_particlePair(Particle *particleI, Particle *particleJ, double a, msm4g_smoothing_handler gamma)
164 {
165  D3Vector rij;
166  double r,r2;
167  double a2;
168  double smoothedkernel,smoothedkernelderivative;
169  double coeff;
170  double potential = 0.0;
171 
172  /* rij = rj - ri */
173  msm4g_d3vector_daxpy(&rij, &(particleJ->r), -1.0, &(particleI->r));
174  /* r^2 = |rij|^2 */
175  r2=msm4g_d3vector_normsquare(&rij);
176  r=sqrt(r2);
177  a2=a*a;
178  /* If only the particles are closer than cut-off distance */
179  if (r2 < a2)
180  {
181  smoothedkernel = 1.0/r - gamma(r/a,0)/a;
182  smoothedkernelderivative = 1.0/r2 - gamma(r/a,1)/a2;
183  coeff = particleI->m * particleJ->m * smoothedkernelderivative/r;
184  /* f_j = f_j + coeff*rij */
185  msm4g_d3vector_daxpy(&(particleJ->fshort) , &(particleJ->fshort), coeff, &rij);
186  /* f_i = f_i - coeff*rij */
187  msm4g_d3vector_daxpy(&(particleI->fshort) , &(particleI->fshort), coeff, &rij);
188  potential = particleI->m * particleJ->m * smoothedkernel;
189  }
190  return potential;
191 }

778 void msm4g_acceleration(double *a,double *r,int n,int d,double *m,double G)
779 {
780  int i,j,k;
781  double rij2,rij3,acc;
782  msm4g_zeros(a,n*d);
783  for (i=0;i<n-1;i++)
784  {
785  for (j=i+1;j<n;j++)
786  {
787  rij2=0.0;
788  for (k=0;k<d;k++)
789  rij2 +=(r[d*i+k]-r[d*j+k])*(r[d*i+k]-r[d*j+k]);
790  rij3=sqrt(rij2)*rij2;
791  for (k=0;k<d;k++)
792  {
793  acc=G*(r[d*i+k]-r[d*j+k])/rij3;
794  a[d*j+k] += m[i]*acc;
795  a[d*i+k] -= m[j]*acc;
796  }
797  }

827  }

829  }
