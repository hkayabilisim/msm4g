1261  double r2i = (rix-rjx-Ax*px)*(rix-rjx-Ax*px) ;

1268  particleI->potential_short_real += 0.5 * particleJ->m * g0;
1269  particleJ->potential_short_real += 0.5 * particleI->m * g0;

1271  double ry = riy - rjy - Ay * py ;
1272  double rz = riz - rjz - Az * pz ;

1280  particleJ->acc_short[2] -= particleI->m * gamaprime * rz ;
1281  }
1282  }
1283  }
1284  } else {

1298  &(particleJ->fshort), coeff, &rij);
1299  /* f_i = f_i - coeff*rij */

1305  }
1306  }
